<script>

let musicPlaying = false;

 

function toggleMusic() {

    const frame = document.getElementById("musicFrame");

    if (!musicPlaying) {

        frame.src += "&autoplay=1";

        musicPlaying = true;

    } else {

        frame.src = frame.src.replace("&autoplay=1", "");

        musicPlaying = false;

    }

}

 

function moveNo() {

    const no = document.getElementById("no");

    no.style.top = y + "px";

}
